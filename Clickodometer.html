class Potion {
    constructor(color, effect) {
        this.color = color;
        this.effect = effect;
    }

    brew() {
        console.log(`The ${this.color} potion is brewing...`);
        if (this.effect === "Invisibility") {
            return "Poof! You vanished.";
        }
        return `It has a magical effect: ${this.effect}`;
    }
}

const fearPotion = new Potion("Midnight Black", "Eternal Dread");
console.log(fearPotion.brew());
